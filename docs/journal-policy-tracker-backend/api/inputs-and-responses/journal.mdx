### Journal Input {#journalInput}

```javascript
JournalInput: {
  title: String;
  url: String;
  issn: String;
  domainName: String;
  policies: PoliciesInput: {
    title: String;
    firstYear: Int;
    lastYear: Int;
    policyType: enum: {
      NUMBER_ONE;
      NUMBER_TWO;
      NUMBER_THREE;
      NUMBER_FOUR;
    }
    isDataAvailabilityStatementPublished: Boolean;
    isDataShared: Boolean;
    isDataPeerReviewed: Boolean;
    enforced: enum: {
      YES;
      NO;
      SOMETIMES;
    }
    enforcedEvidence: String;
  }
}
```

### Journal Response {#journalResponse}

```javascript
JournalResponse: {
  journal: Journal;
  errors: [
    {
      field: String,
      message: String,
    },
  ];
}
```

### Paginated Journals {#paginatedJournals}

```javascript
PaginatedJournals: {
  journals: [Journal];
  totalJournals: Int;
}
```
