### Register Input

```javascript
RegisterInput: {
  fullName: String;
  username: String;
  email: String;
  password: String;
}
```

### Login Input

```javascript
LoginInput: {
  usernameOrEmail: String;
  password: String;
}
```

### User Response

```javascript
UserResponse: {
  user: User;
  errors: [
    {
      field: String,
      message: String,
    },
  ];
}
```

### Paginated User

```javascript
PaginatedUser: {
  users: [User];
  totalUsers: Int;
}
```
